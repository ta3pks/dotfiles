  const { project, tag, limit, offset } = c.req.query();

<task id="4" type="implementation">
<title>Create context route</title>

<action>
Create `memory/src/http/routes/context.ts`:

1. GET /context - Get formatted context for a query
   - Query params: q (query), maxTokens (optional), project (optional)

```typescript
import { Hono } from "hono";
import { MemoryService } from "../../core/service.js";

const context = new Hono();
const service = new MemoryService();

context.get("/", async (c) => {
  const query = c.req.query("q");
  const maxTokens = c.req.query("maxTokens");
  const project = c.req.query("project");
  
  if (!query) {
    return c.json({ success: false, error: "Query required" }, 400);
  }
  
  const result = await service.getContext(query, { maxTokens, project });
  return c.json({ success: true, data: result });
});

export { context };
```
</action>

<task id="6" type="implementation">
<title>Create error middleware</title>

<action>
Create `memory/src/http/middleware/error.ts`:

1. Catch-all error handler
2. Return consistent JSON error format
3. Log errors to stderr

```typescript
import { Context } from "hono";

export async function errorHandler(c: Context, next: () => Promise<void>) {
  try {
    await next();
  } catch (error) {
    console.error("HTTP Error:", error);
    return c.json({
      success: false,
      error: error.message || "Internal server error",
      code: error.code || "INTERNAL_ERROR"
    }, error.status || 500);
  }
}
```
</action>
